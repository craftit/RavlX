Each module has to have a declaration of the module in an appropriate header file. This can be conveniently done using a macro as follows:

#ifndef MODULES_LIST_HH
#define MODULES_LIST_HH

#include "Ravl/Threads/MemModules.hh"

#define FUNC(x) bool x (MemModuleBodyC& module)

FUNC(frame);
//: Extracts frames from video source and conditions them
//!param: Module name - frame
//!param: Input from modules - none

FUNC(shotdet);
//: Shot detector
//!param: Module name - shotBoundary
//!param: Input from modules - frame

FUNC(homogs);
//: Computes homographies between adjacent frames and accumulates them
//!param: Module name - homography
//!param: Input from modules - frame, shotBoundary

FUNC(mosaic);
//: Computes mosaics
//!param: Module name - shotClass
//!param: Input from modules - frame, shotBoundary, homography

#undef FUNC

#endif
